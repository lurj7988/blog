                            # 移动文件

                            将需要挂载的目录中的文件先移动出去

                            cd /home/lurj
# 查看隐藏文件
ls -a
# 移动文件包含隐藏文件
sudo mv *.[^.]* /opt/

                            # 磁盘挂载

                            # 此方法只能临时挂载，系统重启后就失效
sudo mount /dev/sda4 /home/lurj
# 设置开机自动挂载，编辑/etc/fstab文件，添加如下内容
sudo vim /etc/fstab
# /dev/sda4       /home/lurj     ext4    defaults        0 0
# 取消挂载
sudo umount /home/lurj
